```dart
  Future<Device?> _connectViaHotspot(ConnectionInfo peerInfo) async {
    try {
      // Enable hotspot on sender side
      final hotspotResult = await _hotspotService.enableHotspot();
      if (!hotspotResult) {
        throw Exception('Failed to enable hotspot');
      }
      
      // Wait for hotspot to be ready
      await Future.delayed(const Duration(seconds: 3));
      
      final hotspotIp = await _networkManager.getLocalIpAddress();
      
      // In hotspot mode, the sender becomes the access point
      // The receiver should connect to the sender's hotspot IP
      // For now, we'll update the peerInfo to use the hotspot IP
      // The actual connection will be established when the receiver connects
      // This is a simplified fix - proper implementation would require
      // the receiver to actively connect to the sender's hotspot
      
      final device = Device(
        id: peerInfo.code,
        name: peerInfo.deviceName,
        ipAddress: hotspotIp, 
        port: peerInfo.port,
        type: DeviceType.mobile,
        connectionType: ConnectionType.wifi,
        discoveredAt: DateTime.now(),
        isConnected: true,
      );
      
      _discoveredPeers[device.id] = device;
      _peerDiscoveredController.add(device);
      
      return device;
    } catch (e) {
      return null;
    }
  }
```
